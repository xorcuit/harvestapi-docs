```javascript JavaScript
const params = new URLSearchParams({
  group: 'https://www.linkedin.com/groups/1898033/',
  // group: '1898033', // Or group ID alternatively
  page: '1',
});
fetch(`https://api.harvest-api.com/linkedin/post-search?${params.toString()}`, {
  headers: { 'X-API-Key': '<api-key>' },
})
  .then((response) => response.json())
  .then((data) => console.log(data));
```

```python Python
import requests

url = "https://api.harvest-api.com/linkedin/post-search?group=1898033&page=1"

headers = {"X-API-Key": "<api-key>"}

response = requests.request("GET", url, headers=headers)
```
