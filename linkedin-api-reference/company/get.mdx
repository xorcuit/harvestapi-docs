```javascript JavaScript
const params = new URLSearchParams({
  url: 'https://www.linkedin.com/company/google/', // by URL
  // universalName: 'google', // by universalName alternatively
});

fetch(`https://harvestapi.net/api/linkedin/profile?${params.toString()}`, {
  headers: { 'X-API-Key': '<api-key>' },
})
  .then((response) => response.json())
  .then((data) => console.log(data));
```

```bash cURL
curl --request GET \
  --url https://harvestapi.net/api/linkedin/profile?universalName=google \
  --header 'X-API-Key: <api-key>'
```

```python Python
import requests

url = "https://harvestapi.net/api/linkedin/profile?universalName=google"

headers = {"X-API-Key": "<api-key>"}

response = requests.request("GET", url, headers=headers)

print(response.text)
```
