```javascript JavaScript
const params = new URLSearchParams({
  url: 'https://www.linkedin.com/jobs/view/4153069088/', // by URL
  // jobId: '4153069088', // by jobId alternatively
});

fetch(`https://harvestapi.net/api/linkedin/job?${params.toString()}`, {
  headers: { 'X-API-Key': '<api-key>' },
})
  .then((response) => response.json())
  .then((data) => console.log(data));
```

```bash cURL
curl --request GET \
  --url https://harvestapi.net/api/linkedin/job?jobId=4153069088 \
  --header 'X-API-Key: <api-key>'
```

```python Python
import requests

url = "https://harvestapi.net/api/linkedin/job?jobId=4153069088"

headers = {"X-API-Key": "<api-key>"}

response = requests.request("GET", url, headers=headers)

print(response.text)
```
