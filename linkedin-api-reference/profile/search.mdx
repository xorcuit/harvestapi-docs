
```python Python
import requests

url = "https://api.harvest-api.com/linkedin/profile-search?search=Mark&companyId=1441&location=Australia&page=1"

headers = {"X-API-Key": "<api-key>"}

response = requests.request("GET", url, headers=headers)

print(response.text)
```
