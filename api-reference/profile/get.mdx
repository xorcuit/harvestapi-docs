```javascript JavaScript
const params = new URLSearchParams({
  url: 'https://www.linkedin.com/in/williamhgates', // by URL
  // publicIdentifier: 'williamhgates', // by publicIdentifier alternatively
});

fetch(`https://harvestapi.net/api/linkedin/profile?${params.toString()}`, {
  headers: { 'X-API-Key': '<api-key>' },
})
  .then((response) => response.json())
  .then((data) => console.log(data))
  .catch((err) => console.error(err));
```

```bash cURL
curl --request GET \
  --url https://harvestapi.net/api/linkedin/profile?publicIdentifier=williamhgates \
  --header 'X-API-Key: <api-key>'
```

```python Python
import requests

url = "https://harvestapi.net/api/linkedin/profile?publicIdentifier=williamhgates"

headers = {"X-API-Key": "<api-key>"}

response = requests.request("GET", url, headers=headers)

print(response.text)
```
